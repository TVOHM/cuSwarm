An example skeleton program layout for a program that uses cuSwarm

#include "simulation.cuh"
int main()
{
try
{
// Create a simulated swarm - starts with a default configuration object (0 boids)
simulation my_simulation;
// Create a configuration object for this swarm to contrl how the swarm will behave
config my_config;
// Set the values you want
my_config.num_boids = 123; // 123 Boids
my_config.max_position = 500.0f; // In a 500 x 500 x 500 units world
my_config.cell_dim = 14; // That is subdivided into 14 cells in each dimension (thats a 14x14x14 matrix - a cell size of 35~ units)
my_config.skipthink_denominator = 3; // 1/3 of the agents will calculate new steerings on an update call
// Apply the configuration to the simulation
try
{
my_simulation.setConfig(my_config);
}
catch(config_exception ce)
{
// The config was invalid - check the documentation for allowed values
// Error message contains name of first invalid variable found
std::string error = ce.what();
}
// This represents your game loop - in this simple example elapsed is set
// to be the number of miliseconds in between each frame
for(int elapsed = 0, last_frame = time(NULL);; elapsed = time(NULL) - last_frame)
{
// Do your game logic here
// If you want, you may change the configuration of the swarm at any time
config new_config;
new_config.num_boids = 1234;
// Unless you are 100% certain your config is valid, you should wrap it in a try/catch like the example above
my_simulation.setConfig(new_config);
// Update the simulation
my_simulation.update(elapsed);
// Get the new agent positions
host_vector<float4> positions = my_simulation.getPositions();
// Do your drawing here using the posiitons
}
}
catch(cuda_exception ce)
{
// No CUDA compatible device found - your game engine should decide how to handle this situation
}
return 0;
}
